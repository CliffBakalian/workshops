A function has one purpose


  public class Program{
    private class User {
      private string first;
      private string last;
      private int age;
      Users next;
      public User(string f, string l, int age){
        first = f;
        last = l;
        this.age = age;
        this.next = null;
      }
    }
    private User users;
    public Program(){
      users = null;
    }
    
    public User lookup(string name){
      for (User user = users;users !=null;users=users.next){
        if (user.name == name)
          return user;
      }
      return null;
    }
    public void addUser(string f, string l, int a){
      User head = new User(f,l,a);
      head.next = users;
      users = head;
    }
  }
